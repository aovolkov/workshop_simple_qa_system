При протечке - отправить сообщение на все колонки с максимальным звуком

script:

#пример вызова скрипта

proba:
alias: Отправка тестового сообщения на колонку
sequence:
- service: script.yandex_message_target
data:
message: "Просто текст" # текст сообщения
room: [1,2] # указываем на какие колонки отправляем сообщение - в примере отправится на 1 и 2 колонку
vol_lev: 0.2 # уровень звука
# сам скрипт отправки сообщения
yandex_message_target:
# 3 входящих переменных message (текст сообщения), room (множестов номеров колонок), vol_lev (громкость)
# room: 1-кухня, 2-гостиная, 3-детская (это у меня так)
variables:
# ниже задаем колонки на которые хотим отправлять сообщения
alice: {'1':'media_player.yandex_station_ff98f0291fa2735b5061a9ce','2':'media_player.yandex_station_FF98F0292A68213E73719D26','3':'media_player.yandex_station_FF98F029B842D31E654BBF45'}
alias: Отправка сообщения на выбранные колонки
icon: "mdi:text-to-speech"
sequence:
service: tts.yandex_station_say
data_template:
entity_id: >
{%- set values = alice.values() | list -%}
{%- set ns = namespace(counter = 1) -%}
{%- set txt = namespace (txt = '') -%}
{%- for i in room -%}
{%- if ns.counter == 1 -%}
{%- set txt.txt = txt.txt ~ values[i-1] -%}
{%- set ns.counter = 2 -%}
{%- else -%}
{%- set txt.txt = txt.txt ~ ', ' ~ values[i-1] -%}
{%- endif -%}
{%- endfor -%}
{{txt.txt}}
message: "{{ message }}"
options:
volume_level: "{{vol_lev}}"
